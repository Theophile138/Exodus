228// These are the pins for ESP8266 boards
229// Name GPIO NodeMCU Function
230#define PIN_D0 16 // GPIO16 WAKE
231#define PIN_D1 5 // GPIO5 User purpose
232#define PIN_D2 4 // GPIO4 User purpose
233#define PIN_D3 0 // GPIO0 Low on boot means enter FLASH mode
234#define PIN_D4 2 // GPIO2 TXD1 (must be high on boot to go to UART0 FLASH mode)
235#define PIN_D5 14 // GPIO14 HSCLK
236#define PIN_D6 12 // GPIO12 HMISO
237#define PIN_D7 13 // GPIO13 HMOSI RXD2
238#define PIN_D8 15 // GPIO15 HCS TXD0 (must be low on boot to enter UART0 FLASH mode)
239#define PIN_D9 3 // RXD0
240#define PIN_D10 1 // TXD0
241
242#define PIN_MOSI 8 // SD1 FLASH and overlap mode
243#define PIN_MISO 7 // SD0
244#define PIN_SCLK 6 // CLK
245#define PIN_HWCS 0 // D3
246
247#define PIN_D11 9 // SD2
248#define PIN_D12 10 // SD4
